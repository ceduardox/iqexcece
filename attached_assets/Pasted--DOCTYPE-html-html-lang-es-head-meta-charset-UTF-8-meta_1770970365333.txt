<script>
    const field = document.getElementById('game-field');
    const svg = document.getElementById('svg-layer');
    const timerEl = document.getElementById('countdown-timer');
    const statusEl = document.getElementById('status-display');
    const scoreEl = document.getElementById('score');
    const levelEl = document.getElementById('level');
    const progressEl = document.getElementById('progress');
    const failOverlay = document.getElementById('fail-overlay');

    let score = 0;
    let level = 1;
    let nextNode = 1;
    let nodesData = [];
    let gameActive = false;
    let timerInterval;
    let startTime;

    function initLevel() {
        field.querySelectorAll('.node').forEach(n => n.remove());
        svg.innerHTML = '';
        failOverlay.style.display = 'none';
        nextNode = 1;
        nodesData = [];
        gameActive = true;
        
        statusEl.innerText = "SINCERIZANDO...";
        statusEl.style.background = "#e8e8ed";
        statusEl.style.color = "#86868b";

        const numNodes = 2 + level;
        progressEl.innerText = `0/${numNodes}`;

        // LÓGICA DE TIEMPO ACELERADO:
        // Nivel 1: ~5 seg. Nivel 10: ~9 seg (pero para 12 nodos, lo que es mucho más difícil)
        const duration = 2 + (numNodes * 0.9) * Math.pow(0.95, level); 

        for (let i = 1; i <= numNodes; i++) {
            const node = document.createElement('div');
            node.className = 'node';
            node.innerText = i;
            
            const x = Math.random() * (field.clientWidth - 80) + 40;
            const y = Math.random() * (field.clientHeight - 80) + 40;
            node.style.left = `${x}px`;
            node.style.top = `${y}px`;
            
            const trigger = (e) => { e.preventDefault(); handleInteraction(i, x, y, node); };
            node.addEventListener('touchstart', trigger, {passive: false});
            node.addEventListener('mousedown', trigger);

            field.appendChild(node);
            nodesData.push({id: i, x, y});
        }

        startTimer(duration);
    }

    function startTimer(seconds) {
        clearInterval(timerInterval);
        startTime = Date.now();
        timerInterval = setInterval(() => {
            let elapsed = (Date.now() - startTime) / 1000;
            let remaining = Math.max(0, seconds - elapsed);
            timerEl.innerText = remaining.toFixed(1);
            if (remaining <= 0) triggerFail();
        }, 50);
    }

    function handleInteraction(id, x, y, element) {
        if (!gameActive || element.classList.contains('completed')) return;

        if (id === nextNode) {
            element.classList.add('completed');
            progressEl.innerText = `${id}/${nodesData.length}`;
            
            if (nextNode > 1) {
                const prev = nodesData[nextNode - 2];
                const line = document.createElementNS("http://www.w3.org/2000/svg", "line");
                line.setAttribute("x1", prev.x); line.setAttribute("y1", prev.y);
                line.setAttribute("x2", x); line.setAttribute("y2", y);
                svg.appendChild(line);
            }

            if (id === nodesData.length) triggerWin();
            nextNode++;
        } else {
            statusEl.innerText = "FALLA";
            statusEl.style.background = "var(--danger)";
            statusEl.style.color = "#fff";
            setTimeout(() => { if(gameActive) statusEl.innerText = "SINCERIZANDO..."; }, 400);
        }
    }

    function triggerWin() {
        gameActive = false;
        clearInterval(timerInterval);
        statusEl.innerText = "ACERTADO";
        statusEl.style.background = "var(--success)";
        statusEl.style.color = "#fff";
        score += level * 100;
        scoreEl.innerText = score;
        level++;
        levelEl.innerText = level;
        setTimeout(initLevel, 600);
    }

    function triggerFail() {
        gameActive = false;
        clearInterval(timerInterval);
        statusEl.innerText = "ERROR CRÍTICO";
        statusEl.style.background = "var(--danger)";
        statusEl.style.color = "#fff";
        failOverlay.style.display = 'flex';
    }

    function restart() {
        score = 0; level = 1;
        scoreEl.innerText = "0"; levelEl.innerText = "1";
        initLevel();
    }

    window.onload = initLevel;
    window.addEventListener('resize', () => { if(gameActive) initLevel(); });
</script>
