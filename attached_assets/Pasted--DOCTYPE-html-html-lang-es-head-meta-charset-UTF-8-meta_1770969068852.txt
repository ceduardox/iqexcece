<script>
    const field = document.getElementById('game-field');
    const catcher = document.getElementById('neuro-catcher');
    const scoreVal = document.getElementById('score');
    const accVal = document.getElementById('acc');
    const fb = document.getElementById('feedback');

    let score = 0;
    let total = 0;
    let hits = 0;
    let posX = window.innerWidth / 2;

    // Control táctil y mouse suave
    const handleMove = (e) => {
        const clientX = e.touches ? e.touches[0].clientX : e.clientX;
        const rect = field.getBoundingClientRect();
        let x = clientX - rect.left;
        
        // Mantener dentro de bordes
        if (x < 35) x = 35;
        if (x > rect.width - 35) x = rect.width - 35;
        
        posX = x;
        catcher.style.left = `${x}px`;
    };

    field.addEventListener('mousemove', handleMove);
    field.addEventListener('touchmove', handleMove);

    function notify(text, color) {
        fb.innerText = text;
        fb.style.color = color;
        fb.style.opacity = '1';
        setTimeout(() => fb.style.opacity = '0', 250);
    }

    function createImpulse() {
        const impulse = document.createElement('div');
        impulse.className = 'impulse';
        const startX = Math.random() * (field.clientWidth - 60) + 30;
        impulse.style.left = `${startX}px`;
        impulse.style.top = `-20px`;
        field.appendChild(impulse);

        total++;
        let y = -20;
        let speed = 4 + (score / 1000); // Aumento progresivo

        const run = setInterval(() => {
            y += speed;
            impulse.style.top = `${y}px`;

            const rect = field.getBoundingClientRect();
            const targetY = rect.height - 100;

            // Detección de colisión mejorada
            if (y > targetY - 20 && y < targetY + 20) {
                const dist = Math.abs(startX - posX);
                if (dist < 35) {
                    score += 50;
                    hits++;
                    scoreVal.innerText = score;
                    notify('SYNC', 'var(--success)');
                    impulse.remove();
                    clearInterval(run);
                    updateStats();
                }
            }

            if (y > rect.height) {
                notify('LOST', 'var(--danger)');
                impulse.remove();
                clearInterval(run);
                updateStats();
            }
        }, 16);
    }

    function updateStats() {
        const accuracy = Math.round((hits / total) * 100);
        accVal.innerText = `${accuracy}%`;
        document.getElementById('level').innerText = Math.floor(score / 500) + 1;
    }

    // Intervalo de aparición basado en progreso
    let spawnRate = 1200;
    function gameLoop() {
        createImpulse();
        let next = Math.max(400, spawnRate - (score/10));
        setTimeout(gameLoop, next);
    }

    gameLoop();
</script>
